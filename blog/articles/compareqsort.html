compare & qsort

C语言
2025年11月13日 4分钟阅读 约200字

compare格式

比较函数，用于排序（展示为整形数组降序）

int compare(const void* a, const void* b)
{
    return (*(int*)b - *(int*)a);
}

qsort格式

void qsort ( void* base, size_t nitems, size_t size, compare );

简化为：qsort(数组名, 元素个数, sizeof(数组类型), 比较函数);

在不同情况下的使用方法

整数数组

int int_arr[] = {3, 1, 4, 1, 5};
qsort(int_arr, 5, sizeof(int), compare_int);

双精度数组

double double_arr[] = {3.14, 2.71, 1.41};
qsort(double_arr, 3, sizeof(double), compare_double);

字符串数组

char* str_arr[] = {"apple", "banana", "cherry"};
qsort(str_arr, 3, sizeof(char*), compare_string);

结构体数组

typedef struct
{
    char name[20];
    int age;
} Person;

Person people[3] = {{"Alice", 25}, {"Bob", 20}, {"Charlie", 30}};
qsort(people, 3, sizeof(Person), compare_by_age);

具体应用

展示 C 
#include <stdio.h>
#include <stdlib.h>

// 比较函数 - 升序
int compare_asc (const void *a, const void *b) 
{
    return (*(int*)a - *(int*)b);
}

// 比较函数 - 降序  
int compare_desc (const void *a, const void *b) 
{
    return (*(int*)b - *(int*)a);  // 反过来就是降序
}

int main() 
{
    int numbers[] = {5, 2, 8, 1, 9, 3, 7, 4, 6};
    int n = sizeof (numbers) / sizeof (numbers[0]);
    
    printf ("原数组: ");
    for (int i = 0; i < n; i++) 
    {
        printf ("%d ", numbers[i]);
    }
    printf ("\n");
    
    // 升序排序
    qsort (numbers, n, sizeof (int), compare_asc);
    printf ("升序排序: ");
    for (int i = 0; i < n; i++) 
    {
        printf ("%d ", numbers[i]);
    }
    printf ("\n");
    
    // 降序排序
    qsort (numbers, n, sizeof (int), compare_desc);
    printf ("降序排序: ");
    for (int i = 0; i < n; i++) 
    {
        printf ("%d ", numbers[i]);
    }
    printf ("\n");
    
    return 0;
}
